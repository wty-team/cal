def change(mark):  #提取"+","-","*","/"的通用功能
    global Sum
    m=d.index(mark)
    if mark=="*":
        Sum = k[m] * k[m+1]
    elif mark=="/":
        Sum = k[m] / k[m+1]
    elif mark=="+":
        Sum = k[m] + k[m+1]
    elif mark=="-":
        Sum = k[m] - k[m+1]
    k[m]=Sum
    del k[m+1]
    d.remove(d[m])
